### Task 2 — Wire output formats section (L4)
**What:** Replace the static `formats` array in `PlaybookTab` with live Supabase data.

**Exact state to add to PlaybookTab:**
```js
const [formatConfigs,    setFormatConfigs]    = useState([]);
const [loading,          setLoading]          = useState(true);
const [editingId,        setEditingId]        = useState(null);
const [showAdd,          setShowAdd]          = useState(false);
```

**Load on mount:**
```js
useEffect(() => {
  apiGetConfigs(agent.id, "output_format")
    .then(setFormatConfigs)
    .catch(() => showToast("Could not load output formats", "⚠"))
    .finally(() => setLoading(false));
}, [agent.id]);
```
*(Guardrail load merged in Task 3)*

**Handlers — copy exactly from NIGP reference:**
- `handleSetDefault(id)` — PATCH `is_default: true`, then re-fetch full list (not optimistic update — NIGP uses re-fetch to keep server as source of truth)
- `handleToggleSelectable(id, val)` — PATCH + optimistic update
- `handleEdit(updated)` — optimistic update in list
- `handleDelete(id)` — confirm dialog + DELETE + filter from list
- `handleFormatAdded(config)` — if new config `is_default`, zero out existing defaults first; prepend to list; close Add form

**Render:** Replace static format map with:
```jsx
{loading && <div style={{...}}>Loading…</div>}
{!loading && formatConfigs.map(config => (
  <ConfigCard key={config.id} config={config}
    onSetDefault={handleSetDefault}
    onToggleSelectable={handleToggleSelectable}
    onEdit={handleEdit}
    onDelete={handleDelete}
    editingId={editingId}
    setEditingId={setEditingId}
    showToast={showToast}
  />
))}
{!loading && !showAdd && (
  <button onClick={() => setShowAdd(true)} style={{...dashed add button...}}>+ Add New Format</button>
)}
{showAdd && (
  <AddConfigForm type="output_format" agentId={agent.id}
    onSaved={handleFormatAdded}
    onCancel={() => setShowAdd(false)}
    showToast={showToast}
  />
)}
```

**canEdit gate:** Wrap Add button in `{canEdit && ...}`. `canEdit = agent.trainable` — same as ResumeTab. Read-only agents see formats but cannot add/edit/delete.

### Task 3 — Wire guardrails section (L5)
**What:** Replace static guardrail text with live single-record load + save.

**Exact state to add to PlaybookTab (merge with Task 2 state):**
```js
const [guardrailText,    setGuardrailText]    = useState("");
const [guardrailId,      setGuardrailId]      = useState(null);
const [savingGuardrail,  setSavingGuardrail]  = useState(false);
```

**Load — merge into single Promise.all with output_format load:**
```js
useEffect(() => {
  Promise.all([
    apiGetConfigs(agent.id, "output_format"),
    apiGetConfigs(agent.id, "guardrail"),
  ]).then(([formats, guardrails]) => {
    setFormatConfigs(formats);
    const g = guardrails.find(r => r.is_default) || guardrails[0];
    if (g) { setGuardrailText(g.text); setGuardrailId(g.id); }
  }).catch(() => showToast("Could not load playbook configs", "⚠"))
    .finally(() => setLoading(false));
}, [agent.id]);
```

**Save handler:**
```js
const saveGuardrail = async () => {
  setSavingGuardrail(true);
  try {
    if (guardrailId) {
      await apiPatchConfig(guardrailId, { text: guardrailText });
    } else {
      const created = await apiSaveConfig({
        agent_id: agent.id, type: "guardrail", name: "default",
        text: guardrailText, is_default: true, is_user_selectable: false,
      });
      setGuardrailId(created.id);
    }
    showToast("Guardrails saved ✦");
  } catch { showToast("Save failed", "⚠"); }
  setSavingGuardrail(false);
};
```

**Render:** Replace static guardrail display with:
```jsx
{loading
  ? <div style={{...}}>Loading…</div>
  : <>
      <textarea
        value={guardrailText}
        onChange={e => setGuardrailText(e.target.value)}
        rows={10}
        style={{ width:"100%", background:T.paper, border:`1px solid ${T.lineSoft}`,
          padding:"10px 12px", fontFamily:mono, fontSize:11, color:T.ink,
          lineHeight:1.7, resize:"vertical", outline:"none", boxSizing:"border-box" }}
      />
      <div style={{ display:"flex", justifyContent:"space-between", alignItems:"center", marginTop:8 }}>
        <div style={{ fontFamily:mono, fontSize:9, color:T.muted, fontStyle:"italic" }}>
          Single guardrail set · applied to all prompts
        </div>
        <button onClick={saveGuardrail} disabled={savingGuardrail}
          style={{ fontFamily:mono, fontSize:9, color:T.flag, background:"transparent",
            border:`1px solid ${T.flag}`, padding:"5px 14px", cursor:"pointer",
            fontWeight:700, textTransform:"uppercase", letterSpacing:.5 }}>
          {savingGuardrail ? "Saving…" : "Save Guardrails"}
        </button>
      </div>
    </>
}
```

**canEdit gate:** If `!canEdit`, render guardrail text as a read-only `<div>` (same mono style) instead of a `<textarea>`. Hide the Save button.

// S-MIGRATE-05 · PE-04 · Playbook Tab CRUD — Node.js tests
// Categories: A (data shape), B (logic), C (string safety), F (Supabase column alignment), L (live API)

const assert = require("assert");

let passed = 0;
let failed = 0;

function test(name, fn) {
  try { fn(); console.log(`  ✓ ${name}`); passed++; }
  catch(e) { console.error(`  ✗ ${name}: ${e.message}`); failed++; }
}

// ── A. Data shape ─────────────────────────────────────────────────────────────
console.log("\nA. Data shape");

test("output_format config has required fields", () => {
  const config = { id: "abc", name: "Executive Briefing", text: "Return a structured...", is_default: true, is_user_selectable: false, agent_id: "robyn", type: "output_format" };
  assert.ok(config.id, "id required");
  assert.ok(config.name, "name required");
  assert.ok(config.text, "text required");
  assert.strictEqual(typeof config.is_default, "boolean");
  assert.strictEqual(typeof config.is_user_selectable, "boolean");
});

test("guardrail config has required fields", () => {
  const g = { id: "xyz", name: "default", text: "NEVER: ...", is_default: true, is_user_selectable: false, agent_id: "robyn", type: "guardrail" };
  assert.ok(g.id);
  assert.ok(g.text);
  assert.strictEqual(g.type, "guardrail");
  assert.strictEqual(g.is_user_selectable, false);
});

test("empty formats array renders no ConfigCards", () => {
  const formatConfigs = [];
  const cards = formatConfigs.map(c => c.id);
  assert.deepStrictEqual(cards, []);
});

// ── B. Logic ──────────────────────────────────────────────────────────────────
console.log("\nB. Logic");

test("handleFormatAdded: new default zeros out existing defaults", () => {
  const existing = [
    { id: "1", name: "A", is_default: true },
    { id: "2", name: "B", is_default: false },
  ];
  const newConfig = { id: "3", name: "C", is_default: true };
  const updated = newConfig.is_default
    ? existing.map(c => ({ ...c, is_default: false }))
    : existing;
  const result = [...updated, newConfig];
  assert.strictEqual(result.find(c => c.id === "1").is_default, false);
  assert.strictEqual(result.find(c => c.id === "3").is_default, true);
});

test("handleFormatAdded: non-default does not affect existing defaults", () => {
  const existing = [{ id: "1", name: "A", is_default: true }];
  const newConfig = { id: "2", name: "B", is_default: false };
  const updated = newConfig.is_default ? existing.map(c => ({ ...c, is_default: false })) : existing;
  const result = [...updated, newConfig];
  assert.strictEqual(result.find(c => c.id === "1").is_default, true);
});

test("handleEdit: replaces correct config by id", () => {
  const configs = [
    { id: "1", name: "Old Name", text: "old" },
    { id: "2", name: "Other", text: "other" },
  ];
  const updated = { id: "1", name: "New Name", text: "new" };
  const result = configs.map(c => c.id === updated.id ? updated : c);
  assert.strictEqual(result[0].name, "New Name");
  assert.strictEqual(result[1].name, "Other");
});

test("handleDelete: filters config by id", () => {
  const configs = [{ id: "1" }, { id: "2" }, { id: "3" }];
  const result = configs.filter(c => c.id !== "2");
  assert.strictEqual(result.length, 2);
  assert.ok(!result.find(c => c.id === "2"));
});

test("handleToggleSelectable: optimistic update", () => {
  const configs = [{ id: "1", is_user_selectable: false }, { id: "2", is_user_selectable: true }];
  const result = configs.map(c => c.id === "1" ? { ...c, is_user_selectable: true } : c);
  assert.strictEqual(result[0].is_user_selectable, true);
  assert.strictEqual(result[1].is_user_selectable, true);
});

test("guardrail load: prefers is_default record", () => {
  const guardrails = [
    { id: "a", text: "fallback", is_default: false },
    { id: "b", text: "primary", is_default: true },
  ];
  const g = guardrails.find(r => r.is_default) || guardrails[0];
  assert.strictEqual(g.id, "b");
  assert.strictEqual(g.text, "primary");
});

test("guardrail load: falls back to first record if no default", () => {
  const guardrails = [
    { id: "a", text: "only one", is_default: false },
  ];
  const g = guardrails.find(r => r.is_default) || guardrails[0];
  assert.strictEqual(g.id, "a");
});

test("guardrail load: empty array leaves text empty string", () => {
  const guardrails = [];
  const g = guardrails.find(r => r.is_default) || guardrails[0];
  const guardrailText = g ? g.text : "";
  assert.strictEqual(guardrailText, "");
});

// ── C. String safety ──────────────────────────────────────────────────────────
console.log("\nC. String safety");

test("guardrailText defaults to empty string not undefined", () => {
  let guardrailText = "";
  const guardrails = [];
  const g = guardrails.find(r => r.is_default) || guardrails[0];
  if (g) guardrailText = g.text;
  assert.strictEqual(typeof guardrailText, "string");
});

test("config name renders safely when undefined", () => {
  const config = { id: "1", name: undefined, text: "text" };
  const name = config.name || "";
  assert.strictEqual(typeof name, "string");
});

// ── F. Supabase column alignment ──────────────────────────────────────────────
console.log("\nF. Supabase column alignment");

test("apiSaveConfig guardrail payload has correct columns", () => {
  const payload = {
    agent_id: "robyn",
    type: "guardrail",
    name: "default",
    text: "NEVER do X",
    is_default: true,
    is_user_selectable: false,
  };

test("apiPatchConfig guardrail payload has only text", () => {

  assert.ok("text" in patchPayload);
  assert.ok(!("type" in patchPayload), "type should not be in patch payload");

test("apiSaveConfig output_format payload has correct columns", () => {
