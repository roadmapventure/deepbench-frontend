### Task 2 — Wire output formats section (L4)
**What:** Replace the static `formats` array in `PlaybookTab` with live Supabase data.

**Exact state to add to PlaybookTab:**
```js
const [formatConfigs,    setFormatConfigs]    = useState([]);
const [loading,          setLoading]          = useState(true);
```

**Load on mount:**
```js

  apiGetConfigs(agent.id, "output_format")
    .then(setFormatConfigs)
    .catch(() => showToast("Could not load output formats", "⚠"))
    .finally(() => setLoading(false));
```

*(Guardrail load merged in Task 3)*

**Handlers — copy exactly from NIGP reference:**
- `handleSetDefault(id)` — PATCH `is_default: true`, then re-fetch full list (not optimistic update — NIGP uses re-fetch to keep server as source of truth)
- `handleToggleSelectable(id, val)` — PATCH + optimistic update
- `handleEdit(updated)` — optimistic update in list
- `handleDelete(id)` — confirm dialog + DELETE + filter from list
- `handleFormatAdded(config)` — if new config `is_default`, zero out existing defaults first; prepend to list; close Add form

**Render:** Replace static format map with:
```jsx
{loading && <div style={{...}}>Loading…</div>}
{!loading && formatConfigs.map(config => (
  <ConfigCard key={config.id} config={config}
    onSetDefault={handleSetDefault}
    onToggleSelectable={handleToggleSelectable}
    onEdit={handleEdit}
    onDelete={handleDelete}
    editingId={editingId}
    setEditingId={setEditingId}
    showToast={showToast}
  />
))}
{!loading && !showAdd && (
  <button onClick={() => setShowAdd(true)} style={{...dashed add button...}}>+ Add New Format</button>
)}
{showAdd && (
  <AddConfigForm type="output_format" agentId={agent.id}
    onSaved={handleFormatAdded}
    onCancel={() => setShowAdd(false)}
    showToast={showToast}
  />
)}
```

**canEdit gate:** Wrap Add button in `{canEdit && ...}`. `canEdit = agent.trainable` — same as ResumeTab. Read-only agents see formats but cannot add/edit/delete.

### Task 3 — Wire guardrails section (L5) — two autosaving textareas
**What:** Replace the single static guardrail block with two live autosaving textareas: "always" and "never". Each is a separate `guardrail` record in `agent-configs`. Any agent's guardrails can be edited — no `canEdit` gate.

**Data model:** Two records, both `type: "guardrail"`:
- `name: "always"` — "What must [firstName] always do?"
- `name: "never"` — "What must [firstName] never do?"

First blur on an empty textarea creates the record (POST). Subsequent blurs PATCH it. Each textarea manages its own record ID independently.

**Exact state to add to PlaybookTab (merge with Task 2 state):**
```js
const [alwaysText,  setAlwaysText]  = useState("");
const [alwaysId,    setAlwaysId]    = useState(null);
const [neverText,   setNeverText]   = useState("");
const [neverId,     setNeverId]     = useState(null);
```

**Load — merge into single Promise.all with output_format load:**
```js
useEffect(() => {
  Promise.all([
    apiGetConfigs(agent.id, "output_format"),
    apiGetConfigs(agent.id, "guardrail"),
  ]).then(([formats, guardrails]) => {
    setFormatConfigs(formats);
    const always = guardrails.find(r => r.name === "always");
    const never  = guardrails.find(r => r.name === "never");
    if (always) { setAlwaysText(always.text); setAlwaysId(always.id); }
    if (never)  { setNeverText(never.text);   setNeverId(never.id); }
  }).catch(() => showToast("Could not load playbook configs", "⚠"))
    .finally(() => setLoading(false));
}, [agent.id]);
```

**Autosave handler — one function, parameterized:**
```js
const saveGuardrail = async (name, text, id, setId) => {
  try {
    if (id) {
      await apiPatchConfig(id, { text });
    } else {
      const created = await apiSaveConfig({
        agent_id: agent.id, type: "guardrail", name,
        text, is_default: false, is_user_selectable: false,
      });
      setId(created.id);
    }
    showToast("Saved ✦");
  } catch { showToast("Save failed", "⚠"); }
};
```

**Render:** Replace static guardrail display with two stacked sub-cards inside the existing guardrail card:


```js
test("handleFormatAdded: new default zeros out existing defaults", () => {
  const existing = [
    { id: "1", name: "A", is_default: true },
    { id: "2", name: "B", is_default: false },
  ];
  const newConfig = { id: "3", name: "C", is_default: true };
  const updated = newConfig.is_default
    ? existing.map(c => ({ ...c, is_default: false }))
    : existing;
  const result = [...updated, newConfig];
  assert.strictEqual(result.find(c => c.id === "1").is_default, false);
  assert.strictEqual(result.find(c => c.id === "3").is_default, true);
});

test("handleFormatAdded: non-default does not affect existing defaults", () => {
  const existing = [{ id: "1", name: "A", is_default: true }];
  const newConfig = { id: "2", name: "B", is_default: false };
  const updated = newConfig.is_default ? existing.map(c => ({ ...c, is_default: false })) : existing;
  const result = [...updated, newConfig];
  assert.strictEqual(result.find(c => c.id === "1").is_default, true);
});

test("handleEdit: replaces correct config by id", () => {
  const configs = [
    { id: "1", name: "Old Name", text: "old" },
    { id: "2", name: "Other", text: "other" },

  const updated = { id: "1", name: "New Name", text: "new" };
  const result = configs.map(c => c.id === updated.id ? updated : c);
  assert.strictEqual(result[0].name, "New Name");

  assert.ok(!result.find(c => c.id === "2"));

test("handleToggleSelectable: optimistic update", () => {
  const configs = [{ id: "1", is_user_selectable: false }, { id: "2", is_user_selectable: true }];
  const result = configs.map(c => c.id === "1" ? { ...c, is_user_selectable: true } : c);
  assert.strictEqual(result[0].is_user_selectable, true);
```
