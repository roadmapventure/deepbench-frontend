## Section 2: CONTEXT

On initial load of the app, show a full-screen overlay modal that acts as a marketing/orientation splash screen for visitors to `deepbench.roadmapventure.com`. The modal:

- Covers >75% of the screen (80vw, max 960px wide, 88vh tall)
- Has a navy/brass border matching Treasury design
- Shows the dashboard grayed out behind it (dark overlay + blur)
- Displays the DeepBench marketing hero content (headline, subhead, animated counters, CTA)
- Is dismissed **for the session** when the user clicks the CTA or the × button
- CTA click → dismiss + stay on current screen (dashboard `/`)
- × close button → dismiss only, same behavior

Dismiss is stored in `sessionStorage` with key `db_splash_dismissed`. On mount, if that key exists, the component renders nothing.

function animateCounter(id, target, duration) {
  const el = document.getElementById(id);

  let start = 0;
  const step = target / (duration / 16);

    start += step;
    if (start >= target) {
      el.textContent = target;
      clearInterval(timer);
    } else {
      el.textContent = Math.round(start);
    }
  }, 16);
}

const store = {};
const sessionStorage = {
  getItem: (k) => store[k] ?? null,
  setItem: (k, v) => { store[k] = v; },
  removeItem: (k) => { delete store[k]; },
};

const STORAGE_KEY = "db_splash_dismissed";

// ── Category E — sessionStorage logic ────────────────────────────────────────
console.log("\nCategory E — Session Storage");

test("Key absent → splash should show", () => {
  sessionStorage.removeItem(STORAGE_KEY);
  const shouldShow = !sessionStorage.getItem(STORAGE_KEY);
  assert(shouldShow, "Expected splash to show when key absent");
});

test("Key present → splash should not show", () => {
  sessionStorage.setItem(STORAGE_KEY, "1");
  const shouldShow = !sessionStorage.getItem(STORAGE_KEY);
  assert(!shouldShow, "Expected splash to hide when key present");
});

test("dismiss() sets the key", () => {
  sessionStorage.removeItem(STORAGE_KEY);
  // simulate dismiss
  sessionStorage.setItem(STORAGE_KEY, "1");
  assert(sessionStorage.getItem(STORAGE_KEY) === "1", "Key should be set after dismiss");
});

test("dismiss() called twice is idempotent", () => {
  sessionStorage.removeItem(STORAGE_KEY);
  sessionStorage.setItem(STORAGE_KEY, "1");
  sessionStorage.setItem(STORAGE_KEY, "1");
  assert(sessionStorage.getItem(STORAGE_KEY) === "1", "Key should still be '1'");
});

test("Fresh session (removeItem) clears the key", () => {
  sessionStorage.setItem(STORAGE_KEY, "1");
  sessionStorage.removeItem(STORAGE_KEY);
  assert(sessionStorage.getItem(STORAGE_KEY) === null, "Key should be null after remove");
});

// ── Category B — counter animation logic ─────────────────────────────────────
console.log("\nCategory B — Counter Logic");

test("animateCounter reaches target", (done) => {
  // Simulate the counter logic inline
  const target = 41;
  const duration = 100;
  let start = 0;
  const step = target / (duration / 16);
  let finalValue = 0;
  let ticks = 0;
  while (start < target && ticks < 10000) {
    start += step;
    ticks++;
    if (start >= target) {
      finalValue = target;
      break;
    } else {
      finalValue = Math.round(start);
    }
  }
  assert(finalValue === target, `Expected ${target}, got ${finalValue}`);
});

test("animateCounter step is positive for positive target", () => {
  const target = 6;
  const duration = 800;
  const step = target / (duration / 16);
  assert(step > 0, "Step must be positive");
});

test("animateCounter intermediate value is between 0 and target", () => {
  const target = 41;
  const duration = 1400;
  const step = target / (duration / 16);
  const firstTick = Math.round(step);
  assert(firstTick >= 0 && firstTick <= target, `First tick ${firstTick} out of range`);
