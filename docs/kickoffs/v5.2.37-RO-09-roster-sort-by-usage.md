## 2. CONTEXT

On load, the Bench roster grid currently renders agents in the static order they appear in `AGENTS` (the hardcoded data array). This session wires a live usage count per agent from `ai_activity_log` so that agents are sorted top-to-bottom, left-to-right by most-used first.

Sort rule:
- Primary: `usageCount DESC` (highest AI call count first)
- Secondary (tie-break): first name alpha `ASC` (e.g. "Brent" before "Chloe" when both are zero)
- Agents with zero usage (no logged `agent_id` rows) sort after all agents with ≥1 usage, then alpha by first name

No visual changes to the cards themselves — only the render order changes.

```js
// FEATURE: RO-09 — per-agent usage count from ai_activity_log
export function useAgentUsageCounts() {
  const [counts, setCounts] = useState({});

  useEffect(() => {
    supabase
      .from('ai_activity_log')
      .select('agent_id')
      .eq('tenant_id', 'global')
      .not('agent_id', 'is', null)
      .then(({ data, error }) => {
        if (error || !data) return;
        const map = {};
        for (const row of data) {
          map[row.agent_id] = (map[row.agent_id] || 0) + 1;
        }
        setCounts(map);
      });
  }, []);

  return counts;
}
```

```js
// test-S-RO-09.mjs
// Tests the sort logic in isolation — no app imports

function sortAgentsByUsage(agents, usageCounts) {
  return [...agents].sort((a, b) => {
    const ca = usageCounts[a.id] || 0;
    const cb = usageCounts[b.id] || 0;
    if (cb !== ca) return cb - ca;
    return a.name.split(' ')[0].localeCompare(b.name.split(' ')[0]);
  });
}

const agents = [
  { id: 'chloe',   name: 'Chloe Okafor'      },
  { id: 'mike',    name: 'Mike Alvarez'       },
  { id: 'bob',     name: 'Bob Whitfield'      },
  { id: 'brent',   name: 'Brent Matthews'     },
  { id: 'pat',     name: 'Pat Smiley'         },
  { id: 'michelle',name: 'Michelle Manning'   },
  { id: 'dan',     name: 'Dan Bingham'        },
];

let pass = 0;
let fail = 0;

function assert(desc, cond) {
  if (cond) { console.log(`  ✓ ${desc}`); pass++; }
  else       { console.error(`  ✗ ${desc}`); fail++; }
}

// T1: sort by count descending
{
  const counts = { mike: 10, bob: 5, chloe: 8 };
  const sorted = sortAgentsByUsage(agents, counts);
  assert('T1a: mike (10) is first',    sorted[0].id === 'mike');
  assert('T1b: chloe (8) is second',   sorted[1].id === 'chloe');
  assert('T1c: bob (5) is third',      sorted[2].id === 'bob');
}

// T2: alpha tie-break on first name (zero counts)
{
  const counts = {};
  const sorted = sortAgentsByUsage(agents, counts);
  const firstNames = sorted.map(a => a.name.split(' ')[0]);
  assert('T2: all-zero counts → alpha by first name', 
    JSON.stringify(firstNames) === JSON.stringify([...firstNames].sort()));
}

// T3: zero count agents sort after non-zero
{
  const counts = { dan: 3 };
  const sorted = sortAgentsByUsage(agents, counts);
  assert('T3a: dan (3) is first',          sorted[0].id === 'dan');
  assert('T3b: all others have 0 count',   sorted.slice(1).every(a => !counts[a.id]));
}

// T4: tie between two agents → alpha first name
{
  const counts = { mike: 5, chloe: 5 };
  const sorted = sortAgentsByUsage(agents, counts);
  const topTwo = sorted.slice(0, 2).map(a => a.id);
  assert('T4: chloe before mike on tie (C < M alpha)', topTwo[0] === 'chloe' && topTwo[1] === 'mike');
}

// T5: partial counts — agents not in map get 0
{
  const counts = { michelle: 7, pat: 7 };
  const sorted = sortAgentsByUsage(agents, counts);
  assert('T5a: michelle tied with pat — M before P alpha', sorted[0].id === 'michelle');
  assert('T5b: pat is second',                              sorted[1].id === 'pat');
}

console.log(`\n${pass + fail} tests — ${pass} passed, ${fail} failed`);
if (fail > 0) process.exit(1);
else console.log('ALL TESTS PASS');
```
