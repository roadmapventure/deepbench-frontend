## Section 2 — CONTEXT

Two bugs exist in the Prompt Service pipeline that silently corrupt all Prompt Service calls:

**Bug 1 — db-assembly.js: goal never appears in assembled prompt**

`assemblePrompt()` receives `task_context: { goal, deliverable_type }` but `buildSections()` reads only skill profile rows from Supabase. The goal is stored on the returned Prompt Request object but never injected as a prompt section. The LLM therefore never sees the actual work order goal in its system prompt.

Fix: after `buildSections()` returns, inject a `WORK ORDER` section from `task_context.goal` at `order: 2.5` (after intent sections at order 2, before identity at order 3).

**Bug 2 — ai-enrichment.js: `[object Object]` in RAG and Reflect**

`enrichPrompt()` destructures `task_context` from the Prompt Request. That value is the object `{ goal, deliverable_type }`. Two places then use it directly as a string:
- RAG fetch: `queryText: taskContext` → vector search receives `"[object Object]"` → returns irrelevant results
- Reflect prompt: `` ${task_context} `` → execution plan request receives `"[object Object]"` → Reflect produces a generic plan with no task specificity

Fix: immediately after destructuring, derive `taskContextStr`:
```js
const taskContextStr = typeof task_context === 'object' && task_context !== null
  ? (task_context.goal || JSON.stringify(task_context))
  : (task_context || "");
```
Replace all string uses of `task_context` with `taskContextStr`.

// test-s-pm-07a.mjs — S-PM-07a: AA-57 goal injection + stringification fix
// Categories: B (logic), C (string safety)

let passed = 0;
let failed = 0;

function assert(label, condition) {
  if (condition) {
    console.log(`  ✓ ${label}`);
    passed++;
  } else {
    console.error(`  ✗ FAIL: ${label}`);
    failed++;
  }
}

// ── Helpers mirroring the fix logic ──────────────────────────────────────────

function getGoalText(task_context) {
  if (typeof task_context === 'object' && task_context !== null) {
    return task_context.goal || null;
  }
  if (typeof task_context === 'string') return task_context;
  return null;
}

function getTaskContextStr(task_context) {
  if (typeof task_context === 'object' && task_context !== null) {
    return task_context.goal || JSON.stringify(task_context);
  }
  return task_context || "";
}

function buildWorkOrderSection(task_context) {
  const goalText = getGoalText(task_context);
  if (!goalText || !goalText.trim()) return null;
  return {
    slug: 'work-order',
    label: 'WORK ORDER',
    skill_profile_slug: null,
    type: 'stored',
    content: goalText.trim(),
    fetch_instruction: null,
    required: true,
    order: 2.5,
  };
}

// ── Section A: Goal injection (Task 1) ───────────────────────────────────────
console.log('\n[Task 1] db-assembly: WORK ORDER section injection');

{
  const tc = { goal: 'Onboard 3 new NIGP vendors by Q3', deliverable_type: 'execution-plan' };
  const sec = buildWorkOrderSection(tc);
  assert('section created from object task_context', sec !== null);
  assert('slug is work-order', sec?.slug === 'work-order');
  assert('label is WORK ORDER', sec?.label === 'WORK ORDER');
  assert('type is stored', sec?.type === 'stored');
  assert('content equals goal', sec?.content === 'Onboard 3 new NIGP vendors by Q3');
  assert('order is 2.5', sec?.order === 2.5);
  assert('required is true', sec?.required === true);
}

{
  const tc = 'Onboard 3 new NIGP vendors by Q3';
  const sec = buildWorkOrderSection(tc);
  assert('section created from string task_context', sec !== null);
  assert('content equals string goal', sec?.content === 'Onboard 3 new NIGP vendors by Q3');
}

{
  const sec = buildWorkOrderSection({ deliverable_type: 'execution-plan' });
  assert('no section when goal missing from object', sec === null);
}

{
  const sec = buildWorkOrderSection(null);
  assert('no section when task_context is null', sec === null);
}

{
  const sec = buildWorkOrderSection({ goal: '   ' });
  assert('no section when goal is whitespace only', sec === null);
}

// ── Section B: Order after injection ─────────────────────────────────────────
console.log('\n[Task 1] Section ordering after injection');

{
  const mockSections = [
    { slug: 'format', order: 1 },
    { slug: 'intent', order: 2 },
    { slug: 'identity', order: 3 },
    { slug: 'behavior', order: 4 },
  ];
  const sec = buildWorkOrderSection({ goal: 'Test goal' });
  if (sec) mockSections.push(sec);
  mockSections.sort((a, b) => (a.order || 0) - (b.order || 0));
  const slugs = mockSections.map(s => s.slug);
  assert('work-order appears after intent', slugs.indexOf('work-order') > slugs.indexOf('intent'));
  assert('work-order appears before identity', slugs.indexOf('work-order') < slugs.indexOf('identity'));
}

// ── Section C: taskContextStr (Task 2) ───────────────────────────────────────
console.log('\n[Task 2] ai-enrichment: taskContextStr derivation');

{
  const obj = { goal: 'Onboard 3 new NIGP vendors', deliverable_type: 'execution-plan' };
  const str = getTaskContextStr(obj);
  assert('extracts goal string from object', str === 'Onboard 3 new NIGP vendors');
  assert('does not contain [object Object]', !str.includes('[object Object]'));
}

{
  const obj = { deliverable_type: 'execution-plan' };
  const str = getTaskContextStr(obj);
  assert('falls back to JSON when no goal key', str === JSON.stringify(obj));
  assert('fallback does not contain [object Object]', !str.includes('[object Object]'));
}

{
  const str = getTaskContextStr('plain string goal');
  assert('passes through string task_context unchanged', str === 'plain string goal');
}

{
  const str = getTaskContextStr(null);
  assert('null task_context returns empty string', str === '');
}

{
  const str = getTaskContextStr(undefined);
  assert('undefined task_context returns empty string', str === '');
}

{
  const str = getTaskContextStr({ goal: '' });
  assert('empty goal falls back to JSON', typeof str === 'string' && str.length > 0);
}

// ── Summary ───────────────────────────────────────────────────────────────────
console.log(`\n${passed + failed} tests: ${passed} passed, ${failed} failed`);
if (failed > 0) {
  console.error('\n✗ TESTS FAILED — do not commit');
  process.exit(1);
} else {
  console.log('\n✓ ALL TESTS PASS');
}

  const res = await fetch(`${BASE}${path}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(body),
