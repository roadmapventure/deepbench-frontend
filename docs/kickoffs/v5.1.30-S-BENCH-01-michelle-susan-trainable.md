## Section 2 — CONTEXT

Michelle Manning (PP-01) and Susan Smith (TR-08) are both currently `trainable: false` in `agents.js`. This means neither appears as editable in the Personnel File — they can't receive documents via Teach, and their behavioral prompts can't be managed through the UI.

This session does two things:
1. Flips both agents to `trainable: true` in `agents.js`
2. Seeds their base `role_prompt` records in Supabase `agent_configs`

**What this does NOT do:**
- Does NOT wire `api/plan.js` to read Michelle's prompt from Supabase at runtime — that belongs in S-INFRA-01 as part of the unified capability service model
- Does NOT change any API endpoints
- Does NOT change any UI components

After this session, both agents will be editable in the Personnel File (Resume tab, Training tab, Playbook tab) exactly like Bob, Robyn, and Brent.

// Tests: AG-03 + AG-08 — Michelle + Susan trainable flag

import { strict as assert } from "assert";

// ── Inline agent data (mirrors agents.js entries — no app imports) ──────────
const AGENTS = [
  {
    id: "chloe", trainable: false,
  },
  {
    id: "mike", trainable: false,
  },
  {
    id: "bob", trainable: true,
  },
  {
    id: "christy", trainable: false,
  },
  {
    id: "robyn", trainable: true,
  },
  {
    id: "brent", trainable: true,
  },
  {
    id: "pat", trainable: false,
  },
  {
    id: "michelle", trainable: true,   // AG-03: must be true after this session
  },
  {
    id: "susan", trainable: true,      // AG-08: must be true after this session
  },
];

let passed = 0;
let failed = 0;

function test(label, fn) {
  try {
    fn();
    console.log(`  ✓ ${label}`);
    passed++;
  } catch (e) {
    console.error(`  ✗ ${label}: ${e.message}`);
    failed++;
  }
}

console.log("\n── A. Michelle trainable flag ──");
test("michelle.trainable is true", () => {
  const m = AGENTS.find(a => a.id === "michelle");
  assert.ok(m, "michelle not found");
  assert.strictEqual(m.trainable, true, "michelle.trainable must be true");
});

console.log("\n── B. Susan trainable flag ──");
test("susan.trainable is true", () => {
  const s = AGENTS.find(a => a.id === "susan");
  assert.ok(s, "susan not found");
  assert.strictEqual(s.trainable, true, "susan.trainable must be true");
});

console.log("\n── C. Other agents unchanged ──");
const shouldBeTrainable = ["bob", "robyn", "brent"];
const shouldNotBeTrainable = ["chloe", "mike", "christy", "pat"];

for (const id of shouldBeTrainable) {
  test(`${id}.trainable still true`, () => {
    const a = AGENTS.find(a => a.id === id);
    assert.ok(a, `${id} not found`);
    assert.strictEqual(a.trainable, true);
  });
}

for (const id of shouldNotBeTrainable) {
  test(`${id}.trainable still false`, () => {
    const a = AGENTS.find(a => a.id === id);
    assert.ok(a, `${id} not found`);
    assert.strictEqual(a.trainable, false);
  });
}

console.log("\n── D. Supabase seed payload shape ──");
const michellePayload = {
  agent_id: "michelle",
  tenant_id: "global",
  type: "role_prompt",
  name: "Base Planner Prompt",
  text: "You are Michelle Manning, Project Manager at DeepBench.",
  is_default: true,
  is_user_selectable: false,
};

const susanPayload = {
  agent_id: "susan",
  tenant_id: "global",
  type: "role_prompt",
  name: "Base Trainer Prompt",
  text: "You are Susan Smith, Trainer Agent at DeepBench.",
  is_default: true,
  is_user_selectable: false,
};

const requiredKeys = ["agent_id", "tenant_id", "type", "name", "text", "is_default", "is_user_selectable"];

for (const [label, payload] of [["michelle", michellePayload], ["susan", susanPayload]]) {
  test(`${label} seed payload has all required keys`, () => {
    for (const key of requiredKeys) {
      assert.ok(key in payload, `missing key: ${key}`);
      assert.ok(payload[key] !== undefined && payload[key] !== null && payload[key] !== "", `${key} is empty`);
    }
  });
  test(`${label} seed type is role_prompt`, () => {
    assert.strictEqual(payload.type, "role_prompt");
  });
  test(`${label} seed is_default is true`, () => {
    assert.strictEqual(payload.is_default, true);
  });
  test(`${label} seed is_user_selectable is false`, () => {
    assert.strictEqual(payload.is_user_selectable, false);
  });
  test(`${label} seed text starts with agent name`, () => {
    assert.ok(payload.text.startsWith("You are"), `text must start with "You are"`);
  });
}

console.log(`\n${ failed === 0 ? "ALL TESTS PASS" : `${failed} FAILED` } (${passed} passed, ${failed} failed)\n`);
if (failed > 0) process.exit(1);
