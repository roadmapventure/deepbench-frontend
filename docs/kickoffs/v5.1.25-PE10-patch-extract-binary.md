### 2. CONTEXT

**One bug, two files.**

`extractTextFromFile()` in PersonnelScreen.jsx and TeachScreen.jsx was fixed in v5.1.24 to send JSON instead of FormData — correct. But the base64 encoding method chosen was `reader.readAsDataURL(file)` → `split(',')[1]`. For binary files (PDF, DOCX), `readAsDataURL` can silently corrupt the base64 encoding in certain browsers. `api/extract.js` receives corrupted bytes, `pdf-parse` throws, and the endpoint returns 500.

The binary-safe method is `readAsArrayBuffer` → `Uint8Array` → `btoa(binary)`. This is exactly what NIGP's working `extractTextFromFile` uses (`nigp-analyzer/src/TeamBuilder.jsx` line ~244).

No other changes. No new features. No scope creep.

Find the `extractTextFromFile` function (defined above `TrainingTab`, currently using `reader.readAsDataURL`). Replace it entirely with:

```js
async function extractTextFromFile(file) {
  return new Promise((resolve) => {
    const reader = new FileReader();
    reader.onload = async (e) => {
      try {
        const bytes = new Uint8Array(e.target.result);
        let binary = "";
        for (let i = 0; i < bytes.length; i++) binary += String.fromCharCode(bytes[i]);
        const base64 = btoa(binary);
        const res = await fetch("/api/extract", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ fileData: base64, fileType: file.type, fileName: file.name }),
        });
        const data = await res.json();
        if (!res.ok) { resolve({ text: "", wordCount: 0, error: data.error || "Extraction failed" }); return; }
        resolve({ text: data.text, wordCount: data.wordCount, error: null });
      } catch (err) {
        resolve({ text: "", wordCount: 0, error: err.message });
      }
    };
    reader.onerror = () => resolve({ text: "", wordCount: 0, error: "File read failed" });
    reader.readAsArrayBuffer(file);
  });
}
```

test("Uint8Array approach produces valid base64 for known bytes", () => {
  // Simulate what readAsArrayBuffer gives us for the bytes of "Hello"
  const bytes = new Uint8Array([72, 101, 108, 108, 111]); // "Hello"
  let binary = "";
  for (let i = 0; i < bytes.length; i++) binary += String.fromCharCode(bytes[i]);
  const base64 = btoa(binary);
  assert.strictEqual(base64, "SGVsbG8=");
});

test("Uint8Array approach preserves high-byte values (PDF binary)", () => {
  // PDF magic bytes: %PDF- = 0x25 0x50 0x44 0x46 0x2D
  const bytes = new Uint8Array([0x25, 0x50, 0x44, 0x46, 0x2D]);
  let binary = "";
  for (let i = 0; i < bytes.length; i++) binary += String.fromCharCode(bytes[i]);
  const base64 = btoa(binary);
  // Decoding must round-trip perfectly
  const decoded = atob(base64);
  const roundtrip = new Uint8Array(decoded.length);
  for (let i = 0; i < decoded.length; i++) roundtrip[i] = decoded.charCodeAt(i);
  assert.deepStrictEqual(Array.from(roundtrip), [0x25, 0x50, 0x44, 0x46, 0x2D]);
});

test("readAsDataURL split approach would corrupt high bytes (demonstrates the bug)", () => {
  // readAsDataURL produces a data URL like "data:application/pdf;base64,<encoded>"
  // The split(',')[1] approach only works if the encoding is clean ASCII
  // This test shows that for the same bytes, the Uint8Array approach is reliable
  const bytes = new Uint8Array([0xFF, 0xFE, 0x00, 0x01]); // high-byte values common in PDFs
  let binary = "";
  for (let i = 0; i < bytes.length; i++) binary += String.fromCharCode(bytes[i]);
  const base64 = btoa(binary);
  assert.ok(base64.length > 0, "base64 must be non-empty");
  // Verify round-trip integrity
  const decoded = atob(base64);
  assert.strictEqual(decoded.charCodeAt(0), 0xFF);
  assert.strictEqual(decoded.charCodeAt(1), 0xFE);
});

test("base64 output contains only valid base64 characters", () => {
