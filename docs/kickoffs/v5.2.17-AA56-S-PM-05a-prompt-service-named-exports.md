## Section 2 — CONTEXT

S-PM-05a is a pure backend refactor. Three Prompt Service handler files each export only a default handler today. This session extracts the core logic of each into a named export function so that `api/plan.js` (in S-PM-05b) can import and orchestrate the full pipeline without internal HTTP calls.

Additionally:
- `db-assembly.js` gains an optional `runtime_context` string parameter (AA-56). When present, it is injected as an "Additional Context" section at the end of the assembled sections array. This allows the re-generate flow in the Create Work Order screen to pass Q&A clarifying answers into the full Prompt Service pipeline.
- `request-receivable.js` adds `content` to its response object — the parsed plan JSON (steps, questions, title, planSummary, agentId) that the frontend needs to render the plan immediately after generation.

No frontend changes. No new Vercel functions. The existing handler behavior is 100% preserved — callers of the existing endpoints see no change.

```js
// Tests: named exports exist, runtime_context injection, content in response shape

let passed = 0;
let failed = 0;
function test(name, fn) {
  try { fn(); console.log(`  ✓ ${name}`); passed++; }
  catch(e) { console.error(`  ✗ ${name}: ${e.message}`); failed++; }
}
function assert(condition, msg) { if (!condition) throw new Error(msg || 'assertion failed'); }

// ── A: runtime_context injection logic ───────────────────────────────────────
console.log('\nA. runtime_context injection');

test('runtime_context appended when non-empty string', () => {
  const sections = [{ source: 'skill_profiles', type: 'Identity', content: 'Michelle' }];
  const runtime_context = 'User answered: Q1 - internal audience';
  if (runtime_context && typeof runtime_context === 'string' && runtime_context.trim()) {
    sections.push({ source: 'task_context', type: 'Additional Context', content: runtime_context.trim() });
  }
  assert(sections.length === 2, 'section not appended');
  assert(sections[1].type === 'Additional Context', 'wrong section type');
  assert(sections[1].source === 'task_context', 'wrong source');
});

test('runtime_context not appended when null', () => {
  const sections = [{ source: 'skill_profiles', type: 'Identity', content: 'Michelle' }];
  const runtime_context = null;
  if (runtime_context && typeof runtime_context === 'string' && runtime_context.trim()) {
    sections.push({ source: 'task_context', type: 'Additional Context', content: runtime_context.trim() });
  }
  assert(sections.length === 1, 'section appended when null');
});

test('runtime_context not appended when empty string', () => {
  const sections = [{ source: 'skill_profiles', type: 'Identity', content: 'Michelle' }];
  const runtime_context = '   ';
  if (runtime_context && typeof runtime_context === 'string' && runtime_context.trim()) {
    sections.push({ source: 'task_context', type: 'Additional Context', content: runtime_context.trim() });
  }
  assert(sections.length === 1, 'section appended for whitespace-only string');
});

test('runtime_context trimmed before injection', () => {
  const sections = [];
  const runtime_context = '  some context  ';
  if (runtime_context && typeof runtime_context === 'string' && runtime_context.trim()) {
    sections.push({ source: 'task_context', type: 'Additional Context', content: runtime_context.trim() });
  }
  assert(sections[0].content === 'some context', 'content not trimmed');
});

test('runtime_context always last section', () => {
  const sections = [
    { type: 'Format', content: 'a' },
    { type: 'Identity', content: 'b' },
    { type: 'Behavior', content: 'c' },
  ];
  const runtime_context = 'Q1: internal';
  if (runtime_context && typeof runtime_context === 'string' && runtime_context.trim()) {
    sections.push({ source: 'task_context', type: 'Additional Context', content: runtime_context.trim() });
```

```js
// test-s-pm-05a-api.mjs
// Verifies db-assembly, ai-enrichment, and request-receivable endpoints still accept
// their existing request shapes (handler behavior preserved after named export refactor)
// Run: node --env-file=.env.local test-s-pm-05a-api.mjs

const BASE = process.env.VITE_SUPABASE_URL ? 'http://localhost:3000' : 'http://localhost:3000';

async function testDbAssembly() {
  console.log('\nL. DB Assembly endpoint (existing behavior preserved)');
  try {
    const res = await fetch(`${BASE}/api/prompt/db-assembly`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ capability_slug: 'cap-pm-01', agent_id: 'michelle', tenant_id: 'global', task_context: { goal: 'Test plan' } }),
    });
    const data = await res.json();
    if (!data || typeof data !== 'object') throw new Error('no response object');
    console.log('  DB ASSEMBLY API: PASS');
    return true;
  } catch(e) {
    console.error(`  DB ASSEMBLY API: FAIL — ${e.message}`);
    return false;
  }
}

async function testDbAssemblyWithRuntimeContext() {
  console.log('\nL. DB Assembly endpoint with runtime_context');
  try {
    const res = await fetch(`${BASE}/api/prompt/db-assembly`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        capability_slug: 'cap-pm-01',
        agent_id: 'michelle',
        tenant_id: 'global',
        task_context: { goal: 'Test plan' },
        runtime_context: 'Q1: internal audience. Q2: Q3 data only.',
      }),
    });
    const data = await res.json();
    if (!data || typeof data !== 'object') throw new Error('no response object');
    const hasAdditionalContext = JSON.stringify(data).includes('Additional Context');
    if (!hasAdditionalContext) throw new Error('runtime_context not injected into sections');
    console.log('  DB ASSEMBLY runtime_context: PASS');
    return true;
  } catch(e) {
    console.error(`  DB ASSEMBLY runtime_context: FAIL — ${e.message}`);
    return false;
  }
}

async function testRRContent() {
  console.log('\nL. Request & Receivable response includes content field');
  try {
    const dbRes = await fetch(`${BASE}/api/prompt/db-assembly`, {
      method: 'POST', headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ capability_slug: 'cap-pm-01', agent_id: 'michelle', tenant_id: 'global', task_context: { goal: 'Draft a compliance review document' } }),
    });
    const promptRequest = await dbRes.json();

    const enrichRes = await fetch(`${BASE}/api/prompt/ai-enrichment`, {
      method: 'POST', headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ prompt_request: promptRequest, agent_id: 'michelle', capability_slug: 'cap-pm-01' }),
    });
    const enriched = await enrichRes.json();

    const rrRes = await fetch(`${BASE}/api/prompt/request-receivable`, {
      method: 'POST', headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ prompt_request: enriched, agent_id: 'michelle', capability_slug: 'cap-pm-01', tenant_id: 'global' }),
    });
    const result = await rrRes.json();

    if (!result.content) throw new Error('content field missing from R&R response');
    if (!Array.isArray(result.content.steps)) throw new Error('content.steps not array');
    if (result.content.steps.length === 0) throw new Error('content.steps empty');
    console.log(`  R&R content: PASS (${result.content.steps.length} steps returned)`);
    return true;
  } catch(e) {
    console.error(`  R&R content: FAIL — ${e.message}`);
    return false;
  }
}

const results = await Promise.all([testDbAssembly(), testDbAssemblyWithRuntimeContext(), testRRContent()]);
const allPass = results.every(Boolean);
console.log(`\nLive API: ${allPass ? 'ALL PASS' : 'SOME FAILED — do not commit'}`);
if (!allPass) process.exit(1);
```
