## Section 2 — ROOT CAUSE ANALYSIS

### BUG-03: Step cards render empty. Questions render empty. Title missing.

**Execution path:**

1. `sendRequest` in `request-receivable.js` calls Anthropic with `tool_choice: { type: 'tool', name: 'execution-plan' }` and passes the Format Skill schema as the tool's `input_schema`
2. Current schema has `steps: { type: 'array' }` and `questions: { type: 'array' }` — **no `items` definition on either**. No `title` field at all.
3. Anthropic returns tool_use input matching the schema — steps is a valid array, questions is a valid array — but each object inside has whatever fields the LLM guesses from the system prompt
4. `parseResponse` returns `toolUseBlock.input` — shape unknown, fields unknown
5. Screen in `generatePlan()` reads:
   - `p.steps` → `setSteps(newSteps)` → StepCard renders `step.label`, `step.text`, `step.type`, `step.icon` — all undefined → empty cards
   - `p.questions` → `setQuestions(q.q, q.id)` → question text and answer keying → empty question rows
   - `p.title` → `setTitleData` — field not in schema, not returned → title falls back to raw goal text

**Confirmed DB state (queried 2026-06-23):**
```json
{
  "type": "object",
  "required": ["planSummary", "agentId", "agentReason", "steps", "questions"],
  "properties": {
    "steps": {"type": "array"},
    "agentId": {"type": "string"},
    "questions": {"type": "array"},
    "agentReason": {"type": "string"},
    "planSummary": {"type": "string"}
  }
}
```

Missing: `title`, `steps.items`, `questions.items`

### Process gap: Live API test did not catch this

The node test (`test-s-pm-05b-patch.mjs`) only reads source files — it cannot catch runtime failures.
The live API test was a **separate optional file** and appears not to have been run.

**Fix for this session:** The live API test is a single mandatory file. It must pass before committing. It must check actual content of returned steps and questions, not just array length.

```js
// test-s-pm-05b-schema.mjs
// Mandatory live API gate — verifies the execution-plan schema patch produces real content.
// Run: node --env-file=.env.local test-s-pm-05b-schema.mjs
// All tests must pass before committing. This is the QA gate John runs manually.

const BASE = 'http://localhost:3000';
let passed = 0; let failed = 0;

function result(name, ok, detail) {
  if (ok) { console.log(`  ✓ ${name}`); passed++; }
  else { console.error(`  ✗ ${name}${detail ? ': ' + detail : ''}`); failed++; }
}

// ── L1: DB schema verification via db-assembly ────────────────────────────────
console.log('\nL1. DB Assembly — execution-plan schema has items definitions');
try {
  const res = await fetch(`${BASE}/api/prompt/db-assembly`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      capability_slug: 'project-manager',
      agent_id: 'michelle',
      tenant_id: 'global',
      task_context: { goal: 'Test schema verification' },
    }),
  });
  const data = await res.json();
  result('db-assembly returns 200', res.ok, res.status);
  result('format_contract present', !!data.format_contract, 'format_contract missing');
  result('format_contract output_type is json', data.format_contract?.output_type === 'json', `got: ${data.format_contract?.output_type}`);

  const schema = data.format_contract?.schema;
  result('schema present', !!schema, 'schema null or missing');
  result('schema has title field', !!schema?.properties?.title, 'title missing from schema properties');
  result('schema requires title', schema?.required?.includes('title'), `required: ${JSON.stringify(schema?.required)}`);
  result('steps has items definition', !!schema?.properties?.steps?.items, 'steps.items missing — LLM will return empty steps');
  result('steps.items requires label', schema?.properties?.steps?.items?.required?.includes('label'), 'label not required on step items');
  result('steps.items requires text', schema?.properties?.steps?.items?.required?.includes('text'), 'text not required on step items');
  result('steps.items requires type', schema?.properties?.steps?.items?.required?.includes('type'), 'type not required on step items');
  result('questions has items definition', !!schema?.properties?.questions?.items, 'questions.items missing — LLM will return empty questions');
  result('questions.items requires q', schema?.properties?.questions?.items?.required?.includes('q'), 'q not required on question items');
  result('questions.items requires id', schema?.properties?.questions?.items?.required?.includes('id'), 'id not required on question items');
} catch(e) {
  console.error(`  L1 FATAL: ${e.message}`);
  failed++;
}

// ── L2: Full pipeline — real steps with real content ─────────────────────────
console.log('\nL2. Full pipeline — real steps with label, text, type');
let planContent = null;
try {
  const res = await fetch(`${BASE}/api/plan`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      action: 'prompt-service',
      agent_id: 'michelle',
      capability_slug: 'project-manager',
      tenant_id: 'global',
      goal: 'Conduct a compliance review of Q3 vendor contracts to identify risks and policy gaps.',
      deliverable_type: 'compliance-review',
    }),
  });

  if (!res.ok) {
    const text = await res.text();
    result('pipeline returns 200', false, `HTTP ${res.status}: ${text.slice(0, 150)}`);
  } else {
    const data = await res.json();
    planContent = data.content;

    result('response has content field', !!data.content, 'content missing');
    result('content.steps is array', Array.isArray(data.content?.steps), `got: ${typeof data.content?.steps}`);
    result('content.steps not empty', (data.content?.steps?.length || 0) > 0, 'steps array is empty');

    if (data.content?.steps?.length > 0) {
      const step = data.content.steps[0];
      result('step[0].label is non-empty string', typeof step.label === 'string' && step.label.length > 0, `got: ${JSON.stringify(step.label)}`);
      result('step[0].text is non-empty string', typeof step.text === 'string' && step.text.length > 0, `got: ${JSON.stringify(step.text)}`);
      result('step[0].type is agent|human|review', ['agent','human','review'].includes(step.type), `got: ${step.label}`);
      result('step[0].id is a number', typeof step.id === 'number', `got: ${typeof step.id}`);
      console.log(`     → step[0]: id=${step.id}, type="${step.type}", label="${step.label}"`);
    }

    result('content.title is non-empty string', typeof data.content?.title === 'string' && data.content.title.length > 0, `got: ${JSON.stringify(data.content?.title)}`);
    if (data.content?.title) console.log(`     → title: "${data.content.title}"`);

    result('content.planSummary present', typeof data.content?.planSummary === 'string', `got: ${typeof data.content?.planSummary}`);

    if (Array.isArray(data.content?.questions) && data.content.questions.length > 0) {
      const q = data.content.questions[0];
      result('question[0].q is non-empty string', typeof q.q === 'string' && q.q.length > 0, `got: ${JSON.stringify(q.q)}`);
      result('question[0].id is a string', typeof q.id === 'string', `got: ${typeof q.id}`);
      console.log(`     → question[0]: id="${q.id}", q="${q.q?.slice(0,60)}..."`);
    } else {
      result('questions field present (may be empty)', Array.isArray(data.content?.questions), 'questions not an array');
    }

    console.log(`     → ${data.content?.steps?.length} steps returned`);
  }
} catch(e) {
  result('pipeline call succeeded', false, e.message);
}

// ── L3: Suggest-goal still works ──────────────────────────────────────────────
console.log('\nL3. suggest-goal stream still works (regression check)');
try {
  const res = await fetch(`${BASE}/api/plan`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      action: 'suggest-goal',
      agent_id: 'michelle',
      capability_slug: 'project-manager',
      deliverable_label: 'Compliance Review',
      deliverable_description: 'Flags, risks, and policy issues',
      tenant_id: 'global',
    }),
  });
  const contentType = res.headers.get('content-type') || '';
  result('suggest-goal returns SSE stream', contentType.includes('text/event-stream'), `got: ${contentType}`);

  const reader = res.body.getReader();
  const decoder = new TextDecoder();
  let receivedText = false;
  for (let i = 0; i < 15; i++) {
    const { done, value } = await reader.read();
    if (done) break;
    if (decoder.decode(value).includes('"text"')) { receivedText = true; break; }
  }
  reader.cancel();
  result('suggest-goal streams text tokens', receivedText, 'no text tokens received');
} catch(e) {
  result('suggest-goal call succeeded', false, e.message);
}

// ── Summary ───────────────────────────────────────────────────────────────────
const total = passed + failed;
console.log(`\n${total} tests: ${passed} passed, ${failed} failed`);
if (failed > 0) {
  console.error('\nFAIL — do not commit. Fix the schema and re-run.');
  process.exit(1);
} else {
  console.log('\nALL PASS — safe to commit.');
}
```
