## 2. CONTEXT

Priya Nair (CI-02, Forecast/Theory/Performance Expert) already has a static identity in `agents.js` (`AG-21`, shipped `S-APPLE-01a`) and Data Room access already granted in Supabase (`agents.data_room_access = ['apple-cso-data-room']`, confirmed live — same Data Room as Marcus, per design doc §5b: "no separate Data Room per agent"). What she does not yet have is any Skill Profile data or a capability to hold Seniority in — this session builds the first of her two calls.

**Design doc §5.3 (Generate Hypotheses):**
> Trigger: CI-01 (Marcus's) answer flagged `needs_review: true` and the director hasn't already supplied their own claim.
> Input: `{ flagged_question, flagged_answer, review_reason }`
> Output: `{ hypotheses: [ { id, text, rationale } ] }` — 2-4 entries, dynamic

**Why RAG fires on this call even though the design doc's §5b table doesn't list `technical_services: [rag]` for `hyp-generation-intent` specifically (only `hyp-stress-test-intent` lists it):** `hyp-knowledge` (a Knowledge-type Skill Profile) is loaded for *every* call to this capability regardless of which intent is targeted — `db-assembly.js`'s `intent_slug` filter (line 324–326) only excludes non-matching Intent-type profiles, never Knowledge-type ones. This is not a new gap introduced here — it's the exact same behavior `channel-intelligence`'s `ci-routing-intent` already has today (routing doesn't need RAG either, per design doc §5.1, yet `ci-knowledge` still fires a fetch for it), already accepted platform behavior. `SERVICE_CATALOG`'s `patterns` array is capability-level, not per-intent, so this doesn't need a granularity fix here — same as `channel-intelligence`'s entry lists `RAG` even though only one of its two intents uses the result meaningfully.

**Behavior trait content (design doc §5b, already locked):** *"evidentiary rigor calibrated by intent — stricter bar for Correct than Theory"* — this is Priya's general reasoning style across both her calls; at generation time specifically, the emphasis is casting a genuinely wide net across plausible explanations rather than anchoring on one.

// test-s-apple-03a-1-api.mjs — Category A, F, L tests for S-APPLE-03a-1
// Run: node --env-file=.env.local test-s-apple-03a-1-api.mjs
// DELETE BEFORE COMMIT.

const SUPABASE_URL = process.env.SUPABASE_URL;
const SUPABASE_KEY = process.env.SUPABASE_SERVICE_KEY;

let pass = 0, fail = 0;
function check(label, cond) {
  if (cond) { console.log(`PASS: ${label}`); pass++; }
  else { console.log(`FAIL: ${label}`); fail++; }
}

async function supaHeaders() {
  return { "Content-Type": "application/json", "apikey": SUPABASE_KEY, "Authorization": `Bearer ${SUPABASE_KEY}` };
}

// ── Category F: Supabase column alignment — capability + skill profiles + assignment ────────
async function testCategoryF() {
  console.log("\n--- Category F: Supabase column alignment ---");
  const headers = await supaHeaders();

  const capR = await fetch(`${SUPABASE_URL}/rest/v1/capabilities?slug=eq.hypothesis-evaluation&select=slug,execution_type`, { headers });
  const cap = (await capR.json())[0];
  check("hypothesis-evaluation capability exists", !!cap);
  check("hypothesis-evaluation execution_type is ai", cap?.execution_type === 'ai');

  const spR = await fetch(`${SUPABASE_URL}/rest/v1/skill_profiles?slug=in.(hyp-identity,hyp-behavior,hyp-knowledge,hyp-generation-intent)&select=slug,skill_type_slug,llm_model,max_tokens,traits`, { headers });
  const sps = await spR.json();
  check("all 4 hyp-* skill profiles exist", sps.length === 4);
  const byslug = Object.fromEntries(sps.map(s => [s.slug, s]));
  check("hyp-knowledge has traits.broker = 'librarian'", byslug['hyp-knowledge']?.traits?.broker === 'librarian');
  check("hyp-generation-intent has traits.schema.properties.hypotheses", !!byslug['hyp-generation-intent']?.traits?.schema?.properties?.hypotheses);
  check("hyp-generation-intent llm_model is sonnet", byslug['hyp-generation-intent']?.llm_model === 'claude-sonnet-4-6');

  const cspR = await fetch(`${SUPABASE_URL}/rest/v1/capability_skill_profiles?capability_slug=eq.hypothesis-evaluation&select=skill_profile_slug,display_order&order=display_order.asc`, { headers });
  const csp = await cspR.json();
  check("4 capability_skill_profiles rows linked", csp.length === 4);
  check("display_order is 1,2,3,4", JSON.stringify(csp.map(r => r.display_order)) === JSON.stringify([1, 2, 3, 4]));

  const assignR = await fetch(`${SUPABASE_URL}/rest/v1/agent_capability_assignments?agent_id=eq.priya&capability_slug=eq.hypothesis-evaluation`, { headers });
  const assign = await assignR.json();
  check("priya assigned to hypothesis-evaluation", assign.length === 1);
}

  console.log("\n--- Category A: SERVICE_CATALOG data shape ---");
